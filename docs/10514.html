vtest_b   xtlang

Defined in:  https://github.com/digego/extempore/tree/v0.8.9/examples/core/extempore_lang.xtm

Implementation
(bind-func vtest_b
  (let ((iters:i64 80000000)
        (i:i64 0)
        (a:float* (halloc iters))
        (b:float* (halloc iters))
        (c:float* (halloc iters)))
    (dotimes (i iters)
      (pset! a i (i64tof i))
      (pset! b i (i64tof i)))
    (lambda ()
      (dotimes (i (/ iters 4))
        (pset! (cast c /4,float/*) i
               (* (pref (cast a /4,float/*) i)
                  (pref (cast b /4,float/*) i))))
      (if (= (ftoi64 (pref c 80000))
             (* 80000 80000))
          (printf "good sample!\n")
          (printf "bad sample!\n"))
      c)))
