Implementation
(bind-func linear_mc_c
  (let ((time 0.0))
    (lambda (start:SAMPLE end:SAMPLE dur:SAMPLE)
      (let* ((m (/ (- end start) dur))
             (c (- end (* m dur))))
        (lambda (chan incr)
          (if (= chan 0)
              (set! time (+ time incr)))
          (+ (* m time) c))))))
