(bind-func gui_mouse_y
  (lambda ()
    (let ((x_ptr:double* (salloc))
          (y_ptr:double* (salloc)))
      (glfwGetCursorPos (GUI_window EXTEMPORE_GUI) x_ptr y_ptr)
      (convert (pref y_ptr 0) float))))
