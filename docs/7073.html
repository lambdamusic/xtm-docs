(macro do
       (lambda (do-macro)
         (apply (lambda (do vars endtest . body)
                  (let ((do-loop (gensym)))
                    `(letrec ((,do-loop
                               (lambda ,(map (lambda (x)
                                               (if (pair? x) (car x) x))
                                             `,vars)
                                       (if ,(car endtest)
                                           (begin ,@(cdr endtest))
                                           (begin
