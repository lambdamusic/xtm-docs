Implementation
(bind-func div_test:[ADOption*,ADOption*,ADOption*]*
  (lambda (a b)
    (dolet ((n a)  ;; numerator
            (d b)) ;; denominator
           (if (= d 0)
               (ADNone)
               (ADSome (/ n d))))))
