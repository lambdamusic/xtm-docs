(bind-func osc_send_one_double
  (lambda (host ip address a:double)
    (let ((types ",d")
          (typesl 4)
          (mod (modulo (+ (strlen address) 1) 4))
          (addressl (+ (+ (strlen address) 1)
                       (if (= 0 mod) 0 (- 4 mod))))
