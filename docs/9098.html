Defined in:  https://github.com/digego/extempore/tree/v0.8.9/libs/external/portaudio.xtm

Implementation
(bind-func pa_print_devices
  (lambda ()
    (let ((ndev (Pa_GetDeviceCount)))
      (println "Portaudio: found" ndev "devices")
      (doloop (i ndev)
        (println "---- device" i " ----")
        (println (Pa_GetDeviceInfo i))
        (println)))))
