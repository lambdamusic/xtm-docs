(bind-func sys_file_copy
  (lambda (from_path to_path permissions)
    (let ((pool (apr_pool_create))
          (res APR_SUCCESS))
      (set! res (apr_file_copy (cstring from_path) (cstring to_path) permissions pool))
      (apr_pool_destroy pool)
      res)))
