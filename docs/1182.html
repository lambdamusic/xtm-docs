db_load   xtlang

Implementation
(bind-func db_load
  (lambda (filename)
    (let ((db:sqlite3** (salloc))
          (result (sqlite3_open filename db)))
      (if (<> result SQLITE_OK)
          (begin (println "Error opening DB:" (Str (sqlite3_errstr result))) 1)
          (begin (set! DB (pref db 0)) 1))
      void)))
