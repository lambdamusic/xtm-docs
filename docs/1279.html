do   macro

Defined in:  https://github.com/digego/extempore/tree/v0.8.9/runtime/init.xtm

Implementation
(macro do
       (lambda (do-macro)
         (apply (lambda (do vars endtest . body)
                  (let ((do-loop (gensym)))
                    `(letrec ((,do-loop
                               (lambda ,(map (lambda (x)
                                               (if (pair? x) (car x) x))
                                             `,vars)
                                       (if ,(car endtest)
                                           (begin ,@(cdr endtest))
                                           (begin
                                             ,@body
                                             (,do-loop
                                              ,@(map (lambda (x)
                                                       (cond
                                                        ((not (pair? x)) x)
                                                        ((< (length x) 3) (car x))
                                                        (else (car (cdr (cdr x))))))
                                                     `,vars)))))))
                       (,do-loop
                        ,@(map (lambda (x)
                                 (if (and (pair? x) (cdr x))
                                     (car (cdr x))
                                     '()))
                               `,vars)))))
                do-macro)))
