(define loop
  (lambda (time freq dir)
    (change_freq
     (* 1.0 (random 100 500))
     (* 1.0 freq))
    (callback time 'loop (+ time (* 20.0 freq))
              (dir freq 50.0)
              (cond ((> freq 600.0) -)
                    ((< freq 300.0) +)
                    (else dir)))))
