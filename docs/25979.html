(bind-func print:[void,!a]* -> (lambda (r a)
                                 (and (t:vector? a)
                                      (not (t:pointer? a))))
  (lambda (vec)
    (let ((a1 (ref vec)))
      (printout "/" (vref a1 0))
      (doloop (i (- (num-of-elts a1) 1))
        (printout "," (vref a1 (+ 1 i))))
      (printout "/")
      void)))
