(bind-func comb_fb_from_rtsixty
  (let ((fexponent 0.0)
        (rt 0.0))
    (lambda (rt60 delay)
      (set! fexponent (* -3.0 delay (/ 1.0 SRs)))
      (set! rt (* rt60 .001))
      (pow 10.0 (/ fexponent rt)))))
